    function displayDistrictInfo(district) {
        if (districtData[district]) {
            const info = districtData[district];
            document.getElementById('district-info').innerHTML = `
                <h2>${district} District Information</h2>
                <p><strong>Climate:</strong> ${info.climate}</p>
                <p><strong>Soil Type:</strong> ${info.soilType}</p>
                <p><strong>Rainfall:</strong> ${info.rainfall}</p>
                <p><strong>Moisture:</strong> ${info.moisture}</p>
                <p><strong>Trees That Grow Well:</strong> ${info.trees.join(', ')}</p>
            `;
        }
    }

    function recommendPlants() {
        const humidity = document.getElementById('soil-humidity').value;
        const soilType = document.getElementById('soil-type').value;
